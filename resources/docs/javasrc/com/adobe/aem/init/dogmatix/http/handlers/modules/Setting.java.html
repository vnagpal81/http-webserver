package com.adobe.aem.init.dogmatix.http.handlers.modules;

/**
 * Annotation defined to be used as key-value pair within the {@link Module}
 * 
 * @author vnagpal
 *
 */
public @interface Setting {

    /**
     * Key in the settings map
     */
    String name();

    /**
     * Value against the key in the settings map
     */
    String value();
}
