package com.adobe.aem.init.dogmatix.http.handlers.modules;

import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;

/**
 * Indicates that a class is a Module. The server, at startup time, if 
 * instructed via config to scan for Modules in a package, will 
 * reflectively look for classes annotated with this annotation type.
 * Once detected as a module, the class will be instantiated and 
 * initialized using the properties url and settings.
 * 
 * @author vnagpal
 *
 */
@Retention(RetentionPolicy.RUNTIME)
public @interface Module {
    
    /**
     * The URL pattern to match against the request URI to determine 
     * if this module should consume the request.
     * 
     */
    String url();

    /**
     * Detailed module-specific settings required for its functioning. 
     */
    Setting[] settings() default {};

}
